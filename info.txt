def obter_vivos(lista_personagens: list[Personagem]) -> list[Personagem]:
    """Retorna apenas personagens com vida > 0"""
    return [p for p in lista_personagens if p.vida > 0]

def verificar_vitoria(lista_personagens: list[Personagem]) -> bool:
    """Verifica se restou apenas 1 ou nenhum sobrevivente"""
    return len(obter_vivos(lista_personagens)) <= 1

def obter_alvos_validos(atacante: Personagem, 
                        todos: list[Personagem]) -> list[Personagem]:
    """Retorna lista de alvos v√°lidos (exceto o atacante, apenas vivos)"""
    return [p for p in todos if p != atacante and p.vida > 0]

def pode_agir(personagem: Personagem) -> tuple[bool, str]:
    """
    Verifica se personagem pode realizar a√ß√£o
    Retorna: (pode_agir: bool, motivo: str)
    """
    if personagem.esta_congelado():
        return False, "Congelado - n√£o pode agir"
    if personagem.esta_preso():
        return False, "Preso - n√£o pode agir"
    if personagem.esta_fugindo():
        return False, "Em fuga - n√£o pode atacar"
    return True, ""

def obter_habilidades(personagem: Personagem) -> list[str]:
    """
    Retorna lista de habilidades dispon√≠veis para o personagem
    """
    if isinstance(personagem, Fada):
        return ['bomba_de_glitter', 'cristal_de_gelo', 
                'glitter_mortal', 'atacar']
    elif isinstance(personagem, Bardo):
        return ['balada_da_confusao', 'musica_da_cura', 
                'microfonada', 'atacar']
    elif isinstance(personagem, Cozinheiro):
        return ['batata_quente', 'receita_da_mamae', 
                'panela', 'atacar']
    elif isinstance(personagem, Elfo):
        if personagem.classe == "druida":
            return ['se_transforma', 'fuga', 'prissao', 'atacar']
        elif personagem.classe == "arqueiro":
            return ['arco_e_flecha', 'agilidade', 
                    'cristal_de_gelo', 'atacar']
    elif isinstance(personagem, Golem):
        return ['investida', 'regeneracao', 'pele_de_pedra', 'atacar']
    elif isinstance(personagem, Demonio):
        return ['garras_infernais', 'furia', 'drenar_vida', 'atacar']
    elif isinstance(personagem, Assassino):
        return ['apunhalar', 'furtividade', 'golpe_preciso', 'atacar']
    elif isinstance(personagem, Dragao):
        return ['sopro_de_fogo', 'mordida', 'escamas_de_ferro', 
                'voo_rasante', 'rugido_aterrorizante', 'atacar']
    return ['atacar']

def executar_habilidade(atacante: Personagem, 
                        habilidade: str, 
                        alvo: Personagem = None,
                        alvos: list[Personagem] = None) -> str:
    """
    Executa habilidade do atacante
    Retorna: mensagem descritiva da a√ß√£o
    
    Par√¢metros:
        atacante: Quem executa a a√ß√£o
        habilidade: Nome do m√©todo a ser executado
        alvo: Alvo √∫nico (se aplic√°vel)
        alvos: Lista de alvos (para habilidades AOE)
    """
    try:
        metodo = getattr(atacante, habilidade)
        
        # Habilidades AOE que recebem lista
        if habilidade in ['sopro_de_fogo', 'rugido_aterrorizante']:
            if alvos:
                metodo(alvos)
                return f"{atacante.nome} usou {habilidade} (AOE)"
        
        # Habilidades que requerem alvo
        elif alvo and hasattr(metodo, '__code__') and \
             metodo.__code__.co_argcount > 1:
            metodo(alvo)
            return f"{atacante.nome} usou {habilidade} em {alvo.nome}"
        
        # Habilidades sem alvo (self-buff/healing)
        else:
            metodo()
            return f"{atacante.nome} usou {habilidade}"
            
    except Exception as e:
        return f"Erro ao executar {habilidade}: {str(e)}"
